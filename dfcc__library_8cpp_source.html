240 std::set<irep_idt> dfcc_libraryt::get_missing_funs()
241 {
242  std::set<irep_idt> missing;
243 
244  // add `malloc` since it is needed used by the `is_fresh` function
245  missing.insert("malloc");
246 
247  // add `free` and `__CPROVER_deallocate` since they are used by the
248  // `write_set_deallocate_freeable`
249  missing.insert("free");
250 
251  // used by `write_set_release`
252  missing.insert(CPROVER_PREFIX "deallocate");
253 
254  // Make sure all front end functions are loaded
255  missing.insert(CPROVER_PREFIX "assignable");
256  missing.insert(CPROVER_PREFIX "object_from");
257  missing.insert(CPROVER_PREFIX "object_upto");
258  missing.insert(CPROVER_PREFIX "object_whole");
259  missing.insert(CPROVER_PREFIX "freeable");
260 
261  // go over all library functions
262  for(const auto &pair : dfcc_fun_to_name)
263  {
264  symbol_tablet::symbolst::const_iterator found =
265  goto_model.symbol_table.symbols.find(pair.second);
266 
267  if(
268  found == goto_model.symbol_table.symbols.end() ||
269  found->second.value.is_nil())
270  {
271  missing.insert(pair.second);
272  }
273  }
274  return missing;
275 }

279 void dfcc_libraryt::load(std::set<irep_idt> &to_instrument)
280 {
281  PRECONDITION_WITH_DIAGNOSTICS(
282  !loaded, "the cprover_contracts library can only be loaded once");
283  loaded = true;
284 
285  log.status() << "Adding the cprover_contracts library (" << config.ansi_c.arch
286  << ")" << messaget::eom;
287 
288  // these will need to get instrumented as well
289  to_instrument.insert("malloc");
290  to_instrument.insert("free");
291  to_instrument.insert(CPROVER_PREFIX "deallocate");
292 
293  std::set<irep_idt> to_load;
294 
295  // add the whole library
296  to_load.insert(CPROVER_PREFIX "contracts_library");
297 
298  // add front end functions
299  to_load.insert(CPROVER_PREFIX "assignable");
300  to_load.insert(CPROVER_PREFIX "object_from");
301  to_load.insert(CPROVER_PREFIX "object_upto");
302  to_load.insert(CPROVER_PREFIX "object_whole");
303  to_load.insert(CPROVER_PREFIX "freeable");
304 
305  // add stdlib dependences
306  to_load.insert("malloc");
307  to_load.insert("free");
308  to_load.insert(CPROVER_PREFIX "deallocate");
309 
310  symbol_tablet tmp_symbol_table;
311  cprover_c_library_factory_force_load(
312  to_load, tmp_symbol_table, message_handler);
313 
314  // compute missing library functions before modifying the symbol table
315  std::set<irep_idt> missing = get_missing_funs();
316 
317  // copy all loaded symbols to the main symbol table
318  for(const auto &symbol_pair : tmp_symbol_table.symbols)
319  {
320  const auto &sym = symbol_pair.first;
321  if(!goto_model.symbol_table.has_symbol(sym))
322  goto_model.symbol_table.insert(symbol_pair.second);
323  }
324 
325  // compile all missing library functions to GOTO
326  for(const auto &id : missing)
327  {
328  goto_convert(
329  id, goto_model.symbol_table, goto_model.goto_functions, message_handler);
330  }
331 
332  // check that all symbols have a goto_implementation
333  // and populate symbol maps
334  namespacet ns(goto_model.symbol_table);
335  for(const auto &pair : dfcc_fun_to_name)
336  {
337  const auto &found =
338  goto_model.goto_functions.function_map.find(pair.second);
339 
340  INVARIANT(
341  found != goto_model.goto_functions.function_map.end() &&
342  found->second.body_available(),
343  "The body of DFCC library function " + id2string(pair.second) +
344  " could not be found");
345 
346  dfcc_fun_symbol[pair.first] = ns.lookup(pair.second);
347  }
348 
349  // populate symbol maps for easy access to symbols during translation
350  for(const auto &pair : dfcc_type_to_name)
351  {
352  dfcc_type[pair.first] = ns.lookup(pair.second).type;
353  }
354 
355  // fix malloc and free calls
356  fix_malloc_free_calls();
357 
358  // inline the functions that need to be inlined for perf reasons
359  inline_functions();
360 
361  // hide all instructions in counter example traces
362  set_hide(true);
363 }

462 void dfcc_libraryt::fix_malloc_free_calls()
463 {
464  INVARIANT(
465  !malloc_free_fixed,
466  "dfcc_libraryt::fix_malloc_free_calls can only be called once");
467  malloc_free_fixed = true;
468  for(const auto fun : fix_malloc_free_set)
469  {
470  goto_programt &prog =
471  goto_model.goto_functions.function_map.at(dfcc_fun_to_name.at(fun)).body;
472 
473  Forall_goto_program_instructions(ins, prog)
474  {
475  if(ins->is_function_call())
476  {
477  const auto &function = ins->call_function();
478 
479  if(function.id() == ID_symbol)
480  {
481  const irep_idt &fun_name = to_symbol_expr(function).get_identifier();
482 
483  if(fun_name == (CONTRACTS_PREFIX "malloc"))
484  to_symbol_expr(ins->call_function()).set_identifier("malloc");
485  else if(fun_name == (CONTRACTS_PREFIX "free"))
486  to_symbol_expr(ins->call_function()).set_identifier("free");
487  }
488  }
489  }
490  }
