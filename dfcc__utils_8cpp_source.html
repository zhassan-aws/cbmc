289 void dfcc_utilst::clone_parameters(
290  const code_typet::parameterst &old_params,
291  const irep_idt &mode,
292  const irep_idt &module,
293  const source_locationt &location,
294  std::function<const irep_idt(const irep_idt &)> &trans_param,
295  const irep_idt &new_function_id,
296  code_typet::parameterst &new_params)
297 {
298  // rename function parameters in the wrapper function's code_type
299  std::size_t anon_counter = 0;
300 
301  // build parameters and symbols
302  for(auto &old_param : old_params)
303  {
304  // new identifier for new_code_type
305  const irep_idt &old_base_name = old_param.get_base_name().empty()
306  ? "#anon" + std::to_string(anon_counter++)
307  : old_param.get_base_name();
308  const irep_idt &new_base_name = trans_param(old_base_name);
309 
310  irep_idt new_param_id =
311  id2string(new_function_id) + "::" + id2string(new_base_name);
312 
313  // build parameter
314  code_typet::parametert new_param(old_param.type());
315  new_param.set_base_name(new_base_name);
316  new_param.set_identifier(new_param_id);
317  new_params.push_back(new_param);
318 
319  // build symbol
320  parameter_symbolt new_param_symbol;
321  new_param_symbol.base_name = new_base_name;
322  new_param_symbol.name = new_param_id;
323  new_param_symbol.pretty_name = new_param_id;
324  new_param_symbol.type = old_param.type();
325  new_param_symbol.mode = mode;
326  new_param_symbol.module = module;
327  new_param_symbol.location = location;
328  bool add_failed = goto_model.symbol_table.add(new_param_symbol);
329  INVARIANT(
330  !add_failed,
331  "DFCC: renamed parameter " + id2string(new_base_name) +
332  " already exists");
333  }
334 }

359 void dfcc_utilst::wrap_function(
360  const irep_idt &function_id,
361  const irep_idt &wrapped_function_id)
362 {
363  auto &goto_functions = goto_model.goto_functions;
364  auto &symbol_table = goto_model.symbol_table;
365 
366  auto old_function = goto_functions.function_map.find(function_id);
367  INVARIANT(
368  old_function != goto_functions.function_map.end(),
369  "DFCC: function to wrap " + id2string(function_id) +
370  " must exist in the program");
371 
372  // Register the wrapped function under the new name
373  std::swap(
374  goto_functions.function_map[wrapped_function_id], old_function->second);
375 
376  // Remove previous entry
377  goto_functions.function_map.erase(old_function);
378 
379  // Add new symbol for the wrapped function in the symbol table
380  const symbolt *old_sym = symbol_table.lookup(function_id);
381  source_locationt sl(old_sym->location);
382  sl.set_file("wrapped functions for code contracts");
383  sl.set_line("0");
384  symbolt wrapped_sym;
385  wrapped_sym = *old_sym;
386  wrapped_sym.name = wrapped_function_id;
387  wrapped_sym.base_name = wrapped_function_id;
388  wrapped_sym.location = sl;
389  const auto wrapped_found = symbol_table.insert(std::move(wrapped_sym));
390  INVARIANT(
391  wrapped_found.second,
392  "DFCC: wrapped function symbol " + id2string(wrapped_function_id) +
393  " already exists in the symbol table");
394 
395  // Re-insert a symbol for `function_id` which is now the wrapper function
396  symbolt wrapper_sym;
397  wrapper_sym = *old_sym;
398 
399  std::function<const irep_idt(const irep_idt &)> trans_param =
400  [&](const irep_idt &old_param) {
401  return id2string(old_param) + "_wrapper";
402  };
403 
404  // create new code_type with renamed parameters for the wrapper
405  const auto &old_code_type = to_code_type(old_sym->type);
406  code_typet::parameterst new_params;
407  clone_parameters(
408  old_code_type.parameters(),
409  wrapper_sym.mode,
410  wrapper_sym.module,
411  wrapper_sym.location,
412  // the naming scheme is `function_id::param` + `param_suffix`
413  trans_param,
414  function_id,
415  new_params);
416 
417  code_typet new_code_type(new_params, old_code_type.return_type());
418 
419  wrapper_sym.type = new_code_type;
420 
421  // Remove original symbol from the symbol_table
422  bool remove_failed = goto_model.symbol_table.remove(function_id);
423  INVARIANT(
424  !remove_failed,
425  "DFCC: could not remove wrapped function '" + id2string(function_id) +
426  "' from the symbol table");
427 
428  // Insert update symbol in the symbol_table
429  const auto wrapper_sym_found = symbol_table.insert(std::move(wrapper_sym));
430  INVARIANT(
431  wrapper_sym_found.second,
432  "DFCC: could not insert wrapper symbol '" + id2string(function_id) +
433  "' in the symbol table");
434 
435  // Insert wrapper function in the function_map
436  goto_functiont &wrapper_fun = goto_functions.function_map[function_id];
437  wrapper_fun.set_parameter_identifiers(new_code_type);
438  wrapper_fun.body.add(goto_programt::make_end_function(sl));
439 }
