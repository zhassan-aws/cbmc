22  for(const auto &t : loop)

24  // an instruction is the loop end if it is a goto instruction

27  t->is_goto() && t->get_target() == loop_head &&

61 goto_programt::targett get_loop_head_or_end(
62  const unsigned int target_loop_number,
63  goto_functiont &function,
64  bool finding_head)
65 {
66  natural_loops_mutablet natural_loops(function.body);
67 
68  // iterate over all natural loops to find the loop with `target_loop_number`
69  for(const auto &loop_p : natural_loops.loop_map)
70  {
71  const goto_programt::targett loop_head = loop_p.first;
72  goto_programt::targett loop_end =
73  get_loop_end_from_loop_head_and_content_mutable(loop_head, loop_p.second);
74  // check if the current loop is the target loop by comparing loop number
75  if(loop_end->loop_number == target_loop_number)
76  {
77  if(finding_head)
78  return loop_head;
79  else
80  return loop_end;
81  }
82  }
83 
84  UNREACHABLE;
85 }

141 invariant_mapt combine_in_and_post_invariant_clauses(
142  const invariant_mapt &in_clauses,
143  const invariant_mapt &post_clauses,
144  const invariant_mapt &neg_guards)
145 {
146  // Combine invariant
147  // (in_inv || !guard) && (!guard -> pos_inv) for loops with loop guard
148  // in_inv && pos_inv for loops without loop guard
149  invariant_mapt result;
150  for(const auto &in_clause : in_clauses)
151  {
152  const auto &id = in_clause.first;
153  const auto &it_guard = neg_guards.find(id);
154 
155  // Unconditional loop or failed to get loop guard.
156  if(it_guard == neg_guards.end())
157  {
158  result[id] = and_exprt(in_clause.second, post_clauses.at(id));
159  }
160  // Loops with loop guard.
161  else
162  {
163  result[id] = and_exprt(
164  or_exprt(it_guard->second, in_clause.second),
165  implies_exprt(it_guard->second, post_clauses.at(id)));
166  }
167  }
168  return result;
169 }
