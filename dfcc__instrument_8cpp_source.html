353  std::string options = "assert-false-assume-false";

355  auto generate_function_bodies = generate_function_bodies_factory(

423 void dfcc_instrumentt::instrument_instructions(
424  const irep_idt &function_id,
425  const exprt &write_set,
426  goto_programt &goto_program,
427  goto_programt::targett first_instruction,
428  const goto_programt::targett &last_instruction,
429  cfg_infot &cfg_info,
430  const std::function<bool(const goto_programt::targett &)> &pred,
431  std::set<irep_idt> &function_pointer_contracts)
432 {
433  // rewrite pointer_in_range calls
434  dfcc_pointer_in_ranget pointer_in_range(library, message_handler);
435  pointer_in_range.rewrite_calls(
436  goto_program, first_instruction, last_instruction, write_set);
437 
438  // rewrite is_fresh calls
439  dfcc_is_fresht is_fresh(library, message_handler);
440  is_fresh.rewrite_calls(
441  goto_program, first_instruction, last_instruction, write_set);
442 
443  // rewrite is_freeable/was_freed calls
444  dfcc_is_freeablet is_freeable(library, message_handler);
445  is_freeable.rewrite_calls(
446  goto_program, first_instruction, last_instruction, write_set);
447 
448  // rewrite obeys_contract calls
449  dfcc_obeys_contractt obeys_contract(library, message_handler);
450  obeys_contract.rewrite_calls(
451  goto_program,
452  first_instruction,
453  last_instruction,
454  write_set,
455  function_pointer_contracts);
456 
457  const namespacet ns(goto_model.symbol_table);
458  auto &target = first_instruction;
459 
460  // excluding the last
461  while(target != last_instruction)
462  {
463  // Skip instructions marked as disabled for assigns clause checking
464  // or rejected by the predicate
465  if(pred && !pred(target))
466  {
467  target++;
468  continue;
469  }
470 
471  if(target->is_decl())
472  {
473  instrument_decl(function_id, write_set, target, goto_program, cfg_info);
474  }
475  if(target->is_dead())
476  {
477  instrument_dead(function_id, write_set, target, goto_program, cfg_info);
478  }
479  else if(target->is_assign())
480  {
481  instrument_assign(function_id, write_set, target, goto_program, cfg_info);
482  }
483  else if(target->is_function_call())
484  {
485  instrument_function_call(
486  function_id, write_set, target, goto_program, cfg_info);
487  }
488  else if(target->is_other())
489  {
490  instrument_other(function_id, write_set, target, goto_program, cfg_info);
491  }
492  // else do nothing
493  target++;
494  }

607 bool dfcc_instrumentt::must_check_lhs(
608  const source_locationt &lhs_source_location,
609  source_locationt &check_source_location,
610  const irep_idt &language_mode,
611  const exprt &lhs,
612  const cfg_infot &cfg_info)
613 {
614  if(can_cast_expr<symbol_exprt>(lhs))
615  {
616  const auto &symbol_expr = to_symbol_expr(lhs);
617  const auto &id = symbol_expr.get_identifier();
618 
619  check_source_location.set_comment(
620  "Check that " + from_expr_using_mode(ns, language_mode, lhs) +
621  " is assignable");
622 
623  if(cfg_info.is_local(id))
624  return false;
625 
626  // this is a global symbol. Ignore if it is one of the CPROVER globals
627  if(is_cprover_symbol(id))
628  {
629  check_source_location.set_comment(
630  "Check that " + from_expr_using_mode(ns, language_mode, lhs) +
631  " is assignable");
632 
633  return false;
634  }
635 
636  return true;
637  }
638  else
639  {
640  // This is a more complex expression.
641  // Since non-dirty locals are not tracked explicitly in the write set,
642  // we need to skip the check if we can verify that the expression describes
643  // an access to a non-dirty local symbol or an function parameter,
644  // otherwise the check will necessarily fail.
645  // If the expression contains address_of operator,
646  // the assignment gets checked. If the base object of the expression
647  // is a local or a function parameter, it will also be flagged as dirty and
648  // will be tracked explicitly, and the check will pass.
649  // If the expression contains a dereference operation, the assignment gets
650  // checked. If the dereferenced address was computed from a local object,
651  // from a function parameter or returned by a local malloc,
652  // then the object will also be tracked explicitly for being dirty,
653  // and the check will pass.
654  // In all other cases (address of a non-local object, or dereference of
655  // a non-locally computed address) the location must be given explicitly
656  // in the assigns clause to be allowed for assignment and we must check
657  // the assignment.

671 void dfcc_instrumentt::instrument_lhs(
672  const irep_idt &function_id,
673  const exprt &write_set,
674  goto_programt::targett &target,
675  const exprt &lhs,
676  goto_programt &goto_program,
677  cfg_infot &cfg_info)
678 {
679  const auto &mode = utils.get_function_symbol(function_id).mode;
680 
681  goto_programt payload;
682 
683  const auto &lhs_source_location = target->source_location();
684  auto goto_instruction = payload.add(goto_programt::make_incomplete_goto(
685  utils.make_null_check_expr(write_set), lhs_source_location));
686 
687  source_locationt check_source_location(target->source_location());
688  check_source_location.set_property_class("assigns");
689  check_source_location.set_comment(
690  "Check that " + from_expr_using_mode(ns, mode, lhs) + " is assignable");
691 
692  if(must_check_lhs(
693  target->source_location(), check_source_location, mode, lhs, cfg_info))
694  {
695  // ```
696  // IF !write_set GOTO skip_target;
697  // DECL check_assign: bool;
698  // CALL check_assign = check_assignment(write_set, &lhs, sizeof(lhs));
699  // ASSERT(check_assign);
700  // DEAD check_assign;
701  // skip_target: SKIP;
702  // ----
703  // ASSIGN lhs := rhs;
704  // ```
705 
706  auto &check_sym = utils.create_symbol(
707  bool_typet(),
708  id2string(function_id),
709  "__check_lhs_assignment",
710  lhs_source_location,
711  mode,
712  utils.get_function_symbol(function_id).module,
713  false);
714 
715  const auto &check_var = check_sym.symbol_expr();
716 
717  payload.add(goto_programt::make_decl(check_var, lhs_source_location));
718 
719  payload.add(goto_programt::make_function_call(
720  code_function_callt{
721  check_var,
722  library.dfcc_fun_symbol[dfcc_funt::WRITE_SET_CHECK_ASSIGNMENT]
723  .symbol_expr(),
724  {write_set,
725  typecast_exprt::conditional_cast(
726  address_of_exprt(lhs), pointer_type(empty_typet{})),
727  utils.make_sizeof_expr(lhs)}},
728  lhs_source_location));
729 
730  payload.add(
731  goto_programt::make_assertion(check_var, check_source_location));
732  payload.add(goto_programt::make_dead(check_var));
733  }
734  else
735  {
736  // ```
737  // IF !write_set GOTO skip_target;
738  // ASSERT(true);
739  // skip_target: SKIP;
740  // ----
741  // ASSIGN lhs := rhs;
742  // ```
743  payload.add(
744  goto_programt::make_assertion(true_exprt(), check_source_location));
745  }
746 
747  auto label_instruction =
748  payload.add(goto_programt::make_skip(lhs_source_location));
749  goto_instruction->complete_goto(label_instruction);
750 
751  insert_before_swap_and_advance(goto_program, target, payload);
752 }

777 void dfcc_instrumentt::instrument_assign(
778  const irep_idt &function_id,
779  const exprt &write_set,
780  goto_programt::targett &target,
781  goto_programt &goto_program,
782  cfg_infot &cfg_info)
783 {
784  const auto &lhs = target->assign_lhs();
785  const auto &rhs = target->assign_rhs();
786  const auto &target_location = target->source_location();
787 
788  // check the lhs
789  instrument_lhs(function_id, write_set, target, lhs, goto_program, cfg_info);
790 
791  // handle dead_object updates (created by __builtin_alloca for instance)
792  // Remark: we do not really need to track this deallocation since the default
793  // CBMC checks are already able to detect writes to DEAD objects
794  const auto dead_ptr = is_dead_object_update(lhs, rhs);
795  if(dead_ptr.has_value())
796  {
797  // ```
798  // ASSIGN dead_object := if_exprt(nondet, ptr, dead_object);
799  // ----
800  // IF !write_set GOTO skip_target;
801  // CALL record_deallocated(write_set, ptr);
802  // skip_target: SKIP;
803  // ```
804 
805  // step over the instruction
806  target++;
807 
808  goto_programt payload;
809 
810  auto goto_instruction = payload.add(goto_programt::make_incomplete_goto(
811  utils.make_null_check_expr(write_set), target_location));
812 
813  payload.add(goto_programt::make_function_call(
814  code_function_callt{
815  library.dfcc_fun_symbol[dfcc_funt::WRITE_SET_RECORD_DEALLOCATED]
816  .symbol_expr(),
817  {write_set, dead_ptr.value()}},
818  target_location));
819 
820  auto label_instruction =
821  payload.add(goto_programt::make_skip(target_location));
822  goto_instruction->complete_goto(label_instruction);
823 
824  insert_before_swap_and_advance(goto_program, target, payload);
825 
826  // step back
827  target--;
828  }
829 
830  // is the rhs expression a side_effect("allocate") expression ?
831  if(rhs.id() == ID_side_effect && rhs.get(ID_statement) == ID_allocate)
832  {
833  // ```
834  // CALL lhs := side_effect(statemet = ID_allocate, args = {size, clear});
835  // ----
836  // IF !write_set GOTO skip_target;
837  // CALL add_allocated(write_set, lhs);
838  // skip_target: SKIP;
839  // ```
840 
841  // step over the instruction
842  target++;
843 
844  goto_programt payload;
845 
846  auto goto_instruction = payload.add(goto_programt::make_incomplete_goto(
847  utils.make_null_check_expr(write_set), target_location));
848 
849  payload.add(goto_programt::make_function_call(
850  code_function_callt{
851  library.dfcc_fun_symbol[dfcc_funt::WRITE_SET_ADD_ALLOCATED]
852  .symbol_expr(),
853  {write_set, lhs}},
854  target_location));
855 
856  auto label_instruction =
857  payload.add(goto_programt::make_skip(target_location));
858  goto_instruction->complete_goto(label_instruction);
859 
860  insert_before_swap_and_advance(goto_program, target, payload);
861 
862  // step back
863  target--;
864  }

867 void dfcc_instrumentt::instrument_fptr_call_instruction_dynamic_lookup(
868  const exprt &write_set,
869  goto_programt::targett &target,
870  goto_programt &goto_program)
871 {
872  // Insert a dynamic lookup in __instrumented_functions_map
873  // and pass the write set only to functions that are known to be able
874  // to accept it.
875  //
876  // ```
877  // IF __instrumented_functions_map[__CPROVER_POINTER_OBJECT(fptr)] != 1
878  // GOTO no_inst;
879  // CALL [lhs] = fptr(params, write_set);
880  // GOTO end;
881  // no_inst:
882  // CALL [lhs] = fptr(params);
883  // end:
884  // SKIP;
885  // ---
886  // SKIP // [lhs] = fptr(params) turned into SKIP
887  // ```
888  const auto &target_location = target->source_location();
889  const auto &callf = target->call_function();
890  auto object_id = pointer_object(
891  (callf.id() == ID_dereference) ? to_dereference_expr(callf).pointer()
892  : address_of_exprt(callf));
893  auto index_expr = index_exprt(
894  library.get_instrumented_functions_map_symbol().symbol_expr(), object_id);
895  auto cond = notequal_exprt(index_expr, from_integer(1, unsigned_char_type()));
896  goto_programt payload;
897  auto goto_no_inst =
898  payload.add(goto_programt::make_incomplete_goto(cond, target_location));
899  code_function_callt call_inst(
900  target->call_lhs(), target->call_function(), target->call_arguments());
901  call_inst.arguments().push_back(write_set);
902  payload.add(goto_programt::make_function_call(call_inst, target_location));
903  auto goto_end_inst = payload.add(
904  goto_programt::make_incomplete_goto(true_exprt(), target_location));
905  auto no_inst_label = payload.add(goto_programt::make_skip(target_location));
906  goto_no_inst->complete_goto(no_inst_label);
907  code_function_callt call_no_inst(
908  target->call_lhs(), target->call_function(), target->call_arguments());
909  payload.add(goto_programt::make_function_call(call_no_inst, target_location));
910  auto end_label = payload.add(goto_programt::make_skip(target_location));
911  goto_end_inst->complete_goto(end_label);
912  // erase the original call
913  target->turn_into_skip();
914  insert_before_swap_and_advance(goto_program, target, payload);
915 }
