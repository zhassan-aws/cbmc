42 template <bool nondet>
43 optionalt<exprt> expr_initializert<nondet>::expr_initializer_rec(
44  const typet &type,
45  const source_locationt &source_location)
46 {
47  const irep_idt &type_id=type.id();
48 
49  if(type_id==ID_unsignedbv ||
50  type_id==ID_signedbv ||
51  type_id==ID_pointer ||
52  type_id==ID_c_enum ||
53  type_id==ID_c_bit_field ||
54  type_id==ID_bool ||
55  type_id==ID_c_bool ||
56  type_id==ID_floatbv ||
57  type_id==ID_fixedbv)
58  {
59  exprt result;
60  if(nondet)
61  result = side_effect_expr_nondett(type, source_location);
62  else
63  result = from_integer(0, type);
64 
65  result.add_source_location()=source_location;
66  return result;
67  }
68  else if(type_id==ID_rational ||
69  type_id==ID_real)
70  {
71  exprt result;
72  if(nondet)
73  result = side_effect_expr_nondett(type, source_location);
74  else
75  result = constant_exprt(ID_0, type);
76 
77  result.add_source_location()=source_location;
78  return result;
79  }
80  else if(type_id==ID_verilog_signedbv ||
81  type_id==ID_verilog_unsignedbv)
82  {
83  exprt result;
84  if(nondet)
85  result = side_effect_expr_nondett(type, source_location);
86  else
87  {
88  const std::size_t width = to_bitvector_type(type).get_width();
89  std::string value(width, '0');
90 
91  result = constant_exprt(value, type);
92  }
93 
94  result.add_source_location()=source_location;
95  return result;
96  }
97  else if(type_id==ID_complex)
98  {
99  exprt result;
100  if(nondet)
101  result = side_effect_expr_nondett(type, source_location);
102  else
103  {
104  auto sub_zero =
105  expr_initializer_rec(to_complex_type(type).subtype(), source_location);
106  if(!sub_zero.has_value())
107  return {};
108 
109  result = complex_exprt(*sub_zero, *sub_zero, to_complex_type(type));
110  }
111 
112  result.add_source_location()=source_location;
113  return result;
114  }
115  else if(type_id==ID_array)
116  {
117  const array_typet &array_type=to_array_type(type);
118 
119  if(array_type.size().is_nil())
120  {
121  // we initialize this with an empty array
122 
123  array_exprt value({}, array_type);
124  value.type().size() = from_integer(0, size_type());
125  value.add_source_location()=source_location;
126  return std::move(value);
127  }
128  else
129  {
130  auto tmpval =
131  expr_initializer_rec(array_type.element_type(), source_location);
132  if(!tmpval.has_value())
133  return {};
134 
135  const auto array_size = numeric_cast<mp_integer>(array_type.size());
136  if(
137  array_type.size().id() == ID_infinity || !array_size.has_value() ||
138  *array_size > MAX_FLATTENED_ARRAY_SIZE)
139  {
140  if(nondet)
141  return side_effect_expr_nondett(type, source_location);
142 
143  array_of_exprt value(*tmpval, array_type);
144  value.add_source_location()=source_location;
145  return std::move(value);
146  }
147 
148  if(*array_size < 0)
149  return {};
150 
151  array_exprt value({}, array_type);
152  value.operands().resize(
153  numeric_cast_v<std::size_t>(*array_size), *tmpval);
154  value.add_source_location()=source_location;
155  return std::move(value);
156  }
157  }
158  else if(type_id==ID_vector)
159  {
160  const vector_typet &vector_type=to_vector_type(type);
161 
162  auto tmpval =
163  expr_initializer_rec(vector_type.element_type(), source_location);
164  if(!tmpval.has_value())
165  return {};
166 
167  const mp_integer vector_size =
168  numeric_cast_v<mp_integer>(vector_type.size());
169 
170  if(vector_size < 0)
171  return {};
172 
173  vector_exprt value({}, vector_type);
174  value.operands().resize(numeric_cast_v<std::size_t>(vector_size), *tmpval);
175  value.add_source_location()=source_location;
176 
177  return std::move(value);
178  }
179  else if(type_id==ID_struct)
180  {
181  const struct_typet::componentst &components=
182  to_struct_type(type).components();
183 
184  struct_exprt value({}, type);
185 
186  value.operands().reserve(components.size());
187 
188  for(const auto &c : components)
189  {
190  DATA_INVARIANT(
191  c.type().id() != ID_code, "struct member must not be of code type");
192 
193  const auto member = expr_initializer_rec(c.type(), source_location);
194  if(!member.has_value())
195  return {};
196 
197  value.add_to_operands(std::move(*member));
198  }
199 
200  value.add_source_location()=source_location;
201 
202  return std::move(value);
203  }
204  else if(type_id==ID_union)
205  {
206  const union_typet &union_type = to_union_type(type);
207 
208  if(union_type.components().empty())
209  {
210  empty_union_exprt value{type};
211  value.add_source_location() = source_location;
212  return std::move(value);
213  }
214 
215  const auto &widest_member = union_type.find_widest_union_component(ns);
216  if(!widest_member.has_value())
217  return {};
218 
219  auto component_value =
220  expr_initializer_rec(widest_member->first.type(), source_location);
221 
222  if(!component_value.has_value())
223  return {};
224 
225  union_exprt value{widest_member->first.get_name(), *component_value, type};
226  value.add_source_location() = source_location;
227 
228  return std::move(value);
229  }
230  else if(type_id==ID_c_enum_tag)
231  {
232  auto result = expr_initializer_rec(
233  ns.follow_tag(to_c_enum_tag_type(type)), source_location);
234 
235  if(!result.has_value())
236  return {};
237 
238  // use the tag type
239  result->type() = type;
240 
241  return *result;
242  }
243  else if(type_id==ID_struct_tag)
244  {
245  auto result = expr_initializer_rec(
246  ns.follow_tag(to_struct_tag_type(type)), source_location);
247 
248  if(!result.has_value())
249  return {};
250 
251  // use the tag type
252  result->type() = type;
253 
254  return *result;
255  }
256  else if(type_id==ID_union_tag)
257  {
258  auto result = expr_initializer_rec(
259  ns.follow_tag(to_union_tag_type(type)), source_location);
260 
261  if(!result.has_value())
262  return {};
263 
264  // use the tag type
265  result->type() = type;
266 
267  return *result;
268  }
269  else if(type_id==ID_string)
270  {
271  exprt result;
272  if(nondet)
273  result = side_effect_expr_nondett(type, source_location);
274  else
275  result = constant_exprt(irep_idt(), type);
276 
277  result.add_source_location()=source_location;
278  return result;
279  }
280  else
281  return {};
282 }
