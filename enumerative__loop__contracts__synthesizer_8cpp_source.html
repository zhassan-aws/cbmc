46 std::vector<exprt> construct_terminals(const std::set<symbol_exprt> &symbols)
47 {
48  std::vector<exprt> result;
49  for(const auto &e : symbols)
50  {
51  if(e.type().id() == ID_unsignedbv)
52  {
53  // For a variable v with primitive type, we add
54  // v, __CPROVER_loop_entry(v)
55  // into the result.
56  result.push_back(typecast_exprt(e, size_type()));
57  result.push_back(
58  typecast_exprt(unary_exprt(ID_loop_entry, e, e.type()), size_type()));
59  }
60  if((e.type().id() == ID_signedbv))
61  {
62  result.push_back(e);
63  result.push_back(unary_exprt(ID_loop_entry, e, e.type()));
64  }
65  if((e.type().id() == ID_pointer))
66  {
67  // For a variable v with pointer type, we add
68  // __CPROVER_pointer_offset(v),
69  // __CPROVER_pointer_offset(__CPROVER_loop_entry(v))
70  // into the result.
71  result.push_back(pointer_offset_exprt(e, size_type()));
72  result.push_back(pointer_offset_exprt(
73  unary_exprt(ID_loop_entry, e, e.type()), size_type()));
74  }
75  }
76  result.push_back(from_integer(1, signed_short_int_type()));
77  result.push_back(from_integer(0, signed_short_int_type()));
78  return result;
79 }

126 void enumerative_loop_contracts_synthesizert::synthesize_assigns(
127  const exprt &checked_pointer,
128  const std::list<loop_idt> cause_loop_ids)
129 {
130  namespacet ns(goto_model.symbol_table);
131  auto new_assign = checked_pointer;
132 
133  // Add the new assigns target to the most-inner loop that doesn't contain
134  // the new assigns target yet.
135  for(const auto &loop_id : cause_loop_ids)
136  {
137  // Widen index and dereference to whole object.
138  if(new_assign.id() == ID_index || new_assign.id() == ID_dereference)
139  {
140  address_of_exprt address_of_new_assigns(new_assign);
141  havoc_utils_is_constantt is_constant(assigns_map[loop_id], ns);
142  if(!is_constant(address_of_new_assigns))
143  {
144  new_assign = pointer_object(address_of_new_assigns);
145  }
146  }
147 
148  const auto &source_location =
149  get_loop_head(
150  loop_id.loop_number,
151  goto_model.goto_functions.function_map[loop_id.function_id])
152  ->source_location();
153 
154  // Simplify expr to avoid offset that is out of scope.
155  // In the case of nested loops, After widening, pointer_object(ptr + i)
156  // can contain the pointer ptr in the scope of both loops, and the offset
157  // i which is only in the scope of the inner loop.
158  // After simplification, pointer_object(ptr + i) -> pointer_object(ptr).
159  new_assign = simplify_expr(new_assign, ns);
160  new_assign.add_source_location() = source_location;
161 
162  // Avoid adding same target.
163  if(assigns_map[loop_id].insert(new_assign).second)
164  return;
165  }
166  INVARIANT(false, "Failed to synthesize a new assigns target.");
167 }

169 void enumerative_loop_contracts_synthesizert::build_tmp_post_map()
170 {
171  for(auto &goto_function : goto_model.goto_functions.function_map)
172  {
173  for(const auto &instruction : goto_function.second.body.instructions)
174  {
175  // tmp_post variables are symbol lhs of ASSIGN.
176  if(!instruction.is_assign() || instruction.assign_lhs().id() != ID_symbol)
177  continue;
178 
179  const auto symbol_lhs =
180  expr_try_dynamic_cast<symbol_exprt>(instruction.assign_lhs());
181 
182  // tmp_post variables have identifiers with the prefix tmp::tmp_post.
183  if(
184  id2string(symbol_lhs->get_identifier()).find("tmp::tmp_post") !=
185  std::string::npos)
186  {
187  tmp_post_map[instruction.assign_lhs()] = instruction.assign_rhs();
188  }
189  }
190  }

230 exprt enumerative_loop_contracts_synthesizert::synthesize_strengthening_clause(
231  const std::vector<exprt> terminal_symbols,
232  const loop_idt &cause_loop_id,
233  const irep_idt &violation_id)
234 {
235  // Synthesis of strengthening clauses is a enumerate-and-check process.
236  // We first construct the enumerator for the following grammar.
237  // And then enumerate clause and check that if it can make the invariant
238  // inductive.
239 
240  // Initialize factory representing grammar
241  // StartBool -> StartBool && StartBool | Start == Start
242  // | StartBool <= StartBool | StartBool < StartBool
243  // Start -> Start + Start | terminal_symbols
244  // where a0, and a1 are symbol expressions.
245  namespacet ns(goto_model.symbol_table);
246  enumerator_factoryt factory = enumerator_factoryt(ns);
247  recursive_enumerator_placeholdert start_bool_ph(factory, "StartBool", ns);
248  recursive_enumerator_placeholdert start_ph(factory, "Start", ns);
249 
250  // terminals
251  expr_sett leafexprs(terminal_symbols.begin(), terminal_symbols.end());
252 
253  // rules for Start
254  enumeratorst start_args;
255  // Start -> terminals
256  leaf_enumeratort leaf_g(leafexprs, ns);
257  start_args.push_back(&leaf_g);
258 
259  // Start -> Start + Start
260  binary_functional_enumeratort plus_g(
261  ID_plus,
262  start_ph,
263  start_ph,
264  [](const partitiont &partition) {
265  if(partition.size() <= 1)
266  return true;
267  return partition.front() == 1;
268  },
269  ns);
270  start_args.push_back(&plus_g);
271 
272  // rules for StartBool
273  enumeratorst start_bool_args;
274  // StartBool -> StartBool && StartBool
275  binary_functional_enumeratort and_g(ID_and, start_bool_ph, start_bool_ph, ns);
276  start_bool_args.push_back(&and_g);
277  // StartBool -> Start == Start
278  binary_functional_enumeratort equal_g(ID_equal, start_ph, start_ph, ns);
279  start_bool_args.push_back(&equal_g);
280  // StartBool -> Start <= Start
281  binary_functional_enumeratort le_g(ID_le, start_ph, start_ph, ns);
282  start_bool_args.push_back(&le_g);
283  // StartBool -> Start < Start
284  binary_functional_enumeratort lt_g(ID_lt, start_ph, start_ph, ns);
285  start_bool_args.push_back(&lt_g);
286 
287  // add the two nonterminals to the factory
288  factory.attach_productions("Start", start_args);
289  factory.attach_productions("StartBool", start_bool_args);
290 
291  // size of candidates we are searching now,
292  // starting from 0
293  size_t size_bound = 0;
294 
295  // Start to enumerate and check.
296  while(true)
297  {
298  size_bound++;
299 
300  // generate candidate and verify
301  for(auto strengthening_candidate : start_bool_ph.enumerate(size_bound))
302  {
303  log.progress() << "Verifying candidate: "
304  << format(strengthening_candidate) << messaget::eom;
305  invariant_mapt new_in_clauses = invariant_mapt(in_invariant_clause_map);
306  new_in_clauses[cause_loop_id] =
307  and_exprt(new_in_clauses[cause_loop_id], strengthening_candidate);
308  invariant_mapt new_pos_clauses = invariant_mapt(pos_invariant_clause_map);
309  new_pos_clauses[cause_loop_id] =
310  and_exprt(new_pos_clauses[cause_loop_id], strengthening_candidate);
311  const auto &combined_invariant = combine_in_and_post_invariant_clauses(
312  new_in_clauses, new_pos_clauses, neg_guards);
313 
314  // The verifier we use to check current invariant candidates.
315  cegis_verifiert verifier(
316  combined_invariant, assigns_map, goto_model, log);
317 
318  // A good strengthening clause if
319  // 1. all checks pass, or
320  // 2. the loop invariant is inductive and hold upon the entry.
321  const auto &return_cex = verifier.verify();
322  if(
323  !return_cex.has_value() ||
324  (verifier.properties.at(violation_id).status !=
325  property_statust::FAIL &&
326  return_cex->violation_type !=
327  cext::violation_typet::cex_not_hold_upon_entry &&
328  return_cex->violation_type !=
329  cext::violation_typet::cex_not_preserved))
330  {
331  return strengthening_candidate;
332  }
333  }
334  }
335  UNREACHABLE;
336 }

338 invariant_mapt enumerative_loop_contracts_synthesizert::synthesize_all()
339 {
340  init_candidates();
341  build_tmp_post_map();
342 
343  // The invariants we are going to synthesize and verify are the combined
344  // invariants from in- and post- invariant clauses.
345  auto combined_invariant = combine_in_and_post_invariant_clauses(
346  in_invariant_clause_map, pos_invariant_clause_map, neg_guards);
347 
348  // The verifier we use to check current invariant candidates.
349  cegis_verifiert verifier(combined_invariant, assigns_map, goto_model, log);
350 
351  // Set of symbols the violation may be dependent on.
352  // We enumerate strengthening clauses built from symbols from the set.
353  std::set<symbol_exprt> dependent_symbols;
354  // Set of symbols we used to enumerate strengthening clauses.
355  std::vector<exprt> terminal_symbols;
356 
357  log.debug() << "Start the first synthesis iteration." << messaget::eom;
358  auto return_cex = verifier.verify();
359 
360  while(return_cex.has_value())
361  {
362  exprt new_invariant_clause = true_exprt();
363  // Synthesize the new_clause
364  // We use difference strategies for different type of violations.
365  switch(return_cex->violation_type)
366  {
367  case cext::violation_typet::cex_out_of_boundary:
368  new_invariant_clause =
369  synthesize_range_predicate(return_cex->violated_predicate);
370  break;
371 
372  case cext::violation_typet ::cex_null_pointer:
373  new_invariant_clause =
374  synthesize_same_object_predicate(return_cex->checked_pointer);
375  break;
376 
377  case cext::violation_typet::cex_assignable:
378  synthesize_assigns(
379  return_cex->checked_pointer, return_cex->cause_loop_ids);
380  break;
381 
382  case cext::violation_typet::cex_other:
383  // Update the dependent symbols.
384  dependent_symbols = compute_dependent_symbols(
385  return_cex->cause_loop_ids.front(),
386  new_invariant_clause,
387  return_cex->live_variables);
388  case cext::violation_typet::cex_not_preserved:
389  terminal_symbols = construct_terminals(dependent_symbols);
390  new_invariant_clause = synthesize_strengthening_clause(
391  terminal_symbols,
392  return_cex->cause_loop_ids.front(),
393  verifier.target_violation_id);
394  break;
395 
396  case cext::violation_typet::cex_not_hold_upon_entry:
397  INVARIANT(false, "unsupported violation type");
398  break;
399  }
400 
401  // Assigns map has already been updated in the switch block.
402  // Update invariants map for other types of violations.
403  if(return_cex->violation_type != cext::violation_typet::cex_assignable)
404  {
405  INVARIANT(!return_cex->cause_loop_ids.empty(), "No cause loop found!");
406  INVARIANT(
407  new_invariant_clause != true_exprt(),
408  "failed to synthesized meaningful clause");
409 
410  // There could be tmp_post variables in the synthesized clause.
411  // We substitute them with their original variables.
412  replace_tmp_post(new_invariant_clause, tmp_post_map);
413 
414  const auto &cause_loop_id = return_cex->cause_loop_ids.front();
415  // Update the dependent symbols.
416  dependent_symbols = compute_dependent_symbols(
417  cause_loop_id, new_invariant_clause, return_cex->live_variables);
418 
419  // add the new clause to the candidate invariants.
420  if(
421  return_cex->violation_location ==
422  cext::violation_locationt::in_condition)
423  {
424  // When the violation happens in the loop guard, the new clause
425  // should hold for the both cases of
426  // 1. loop guard holds --- loop_guard -> in_invariant
427  // 2. loop guard doesn't hold --- !loop_guard -> pos_invariant
428  in_invariant_clause_map[cause_loop_id] = and_exprt(
429  in_invariant_clause_map[cause_loop_id], new_invariant_clause);
430  pos_invariant_clause_map[cause_loop_id] = and_exprt(
431  pos_invariant_clause_map[cause_loop_id], new_invariant_clause);
432  }
433  else if(
434  return_cex->violation_location == cext::violation_locationt::in_loop)
435  {
436  // When the violation happens in the loop body, the new clause
437  // should hold for the case of
438  // loop guard holds --- loop_guard -> in_invariant
439  in_invariant_clause_map[cause_loop_id] = and_exprt(
440  in_invariant_clause_map[cause_loop_id], new_invariant_clause);
441  }
442  else
443  {
444  // When the violation happens after the loop body, the new clause
445  // should hold for the case of
446  // loop guard doesn't hold --- !loop_guard -> pos_invariant
447  pos_invariant_clause_map[cause_loop_id] = and_exprt(
448  pos_invariant_clause_map[cause_loop_id], new_invariant_clause);
449  }
450 
451  // Re-combine invariant clauses and update the candidate map.
452  combined_invariant = combine_in_and_post_invariant_clauses(
453  in_invariant_clause_map, pos_invariant_clause_map, neg_guards);
454  }
455 
456  return_cex = verifier.verify();
457  }
458 
459  log.result() << "result : " << log.green << "PASS" << log.reset
460  << messaget::eom;
461 
462  return combined_invariant;
463 }
