183 bool static_analysis_baset::visit(
184  const irep_idt &function_identifier,
185  locationt l,
186  working_sett &working_set,
187  const goto_programt &goto_program,
188  const goto_functionst &goto_functions)
189 {
190  bool new_data=false;
191 
192  statet &current=get_state(l);
193 
194  current.seen=true;
195 
196  for(const auto &to_l : goto_program.get_successors(l))
197  {
198  if(to_l==goto_program.instructions.end())
199  continue;
200 
201  std::unique_ptr<statet> tmp_state(
202  make_temporary_state(current));
203 
204  statet &new_values=*tmp_state;
205 
206  if(l->is_function_call())
207  {
208  // this is a big special case
209  do_function_call_rec(
210  function_identifier,
211  l,
212  to_l,
213  l->call_function(),
214  l->call_arguments(),
215  new_values,
216  goto_functions);
217  }
218  else
219  new_values.transform(
220  ns, function_identifier, l, function_identifier, to_l);
221 
222  statet &other=get_state(to_l);
223 
224  bool have_new_values=
225  merge(other, new_values, to_l);
226 
227  if(have_new_values)
228  new_data=true;
229 
230  if(have_new_values || !other.seen)
231  put_in_working_set(working_set, to_l);
232  }
233 
234  return new_data;
235 }

237 void static_analysis_baset::do_function_call(
238  const irep_idt &calling_function,
239  locationt l_call,
240  locationt l_return,
241  const goto_functionst &goto_functions,
242  const goto_functionst::function_mapt::const_iterator f_it,
243  const exprt::operandst &,
244  statet &new_state)
245 {
246  const goto_functionst::goto_functiont &goto_function=f_it->second;
247 
248  if(!goto_function.body_available())
249  return; // do nothing
250 
251  CHECK_RETURN(!goto_function.body.instructions.empty());
252 
253  {
254  // get the state at the beginning of the function
255  locationt l_begin=goto_function.body.instructions.begin();
256 
257  // do the edge from the call site to the beginning of the function
258  std::unique_ptr<statet> tmp_state(make_temporary_state(new_state));
259  tmp_state->transform(ns, calling_function, l_call, f_it->first, l_begin);
260 
261  statet &begin_state=get_state(l_begin);
262 
263  bool new_data=false;
264 
265  // merge the new stuff
266  if(merge(begin_state, *tmp_state, l_begin))
267  new_data=true;
268 
269  // do each function at least once
270  if(functions_done.find(f_it->first)==
271  functions_done.end())
272  {
273  new_data=true;
274  functions_done.insert(f_it->first);
275  }
276 
277  // do we need to do the fixedpoint of the body?
278  if(new_data)
279  {
280  // recursive call
281  fixedpoint(f_it->first, goto_function.body, goto_functions);
282  }
283  }
284 
285  {
286  // get location at end of procedure
287  locationt l_end=--goto_function.body.instructions.end();
288 
289  DATA_INVARIANT(l_end->is_end_function(), "must be end of function");
290 
291  statet &end_of_function=get_state(l_end);
292 
293  // do edge from end of function to instruction after call
294  std::unique_ptr<statet> tmp_state(
295  make_temporary_state(end_of_function));
296  tmp_state->transform(ns, f_it->first, l_end, calling_function, l_return);
297 
298  // propagate those -- not exceedingly precise, this is,
299  // as still it contains all the state from the
300  // call site
301  merge(new_state, *tmp_state, l_return);
302  }
303 
304  {
305  // effect on current procedure (if any)
306  new_state.transform(
307  ns, calling_function, l_call, calling_function, l_return);
308  }

311 void static_analysis_baset::do_function_call_rec(
312  const irep_idt &calling_function,
313  locationt l_call,
314  locationt l_return,

321  if(goto_functions.function_map.empty())
322  return;
323

326  const irep_idt &identifier = to_symbol_expr(function).get_identifier();
327 
328  if(recursion_set.find(identifier)!=recursion_set.end())
329  {
330  // recursion detected!

341 
342  do_function_call(
343  calling_function,
344  l_call,
345  l_return,
346  goto_functions,
347  it,
348  arguments,
349  new_state);
350 
351  recursion_set.erase(identifier);
352  }
353  else if(function.id()==ID_if)
354  {
355  if(function.operands().size()!=3)
356  throw "if takes three arguments";
357 
358  std::unique_ptr<statet> n2(make_temporary_state(new_state));
359 
360  do_function_call_rec(
361  calling_function,
362  l_call,
363  l_return,

365  arguments,
366  new_state,
367  goto_functions);

370  calling_function,
371  l_call,
372  l_return,
373  to_if_expr(function).false_case(),
374  arguments,
375  *n2,
376  goto_functions);
377 
378  merge(new_state, *n2, l_return);

400  arguments,
401  *n2,
402  goto_functions);
403  merge(new_state, *n2, l_return);
404  }
405  }
406  }

410  // ignore, can't be a function
411  }
412  else if(function.id()==ID_member || function.id()==ID_index)

440  const goto_functionst &goto_functions)
441 {

443 
444  is_threadedt is_threaded(goto_functions);
445 
446  // construct an initial shared state collecting the results of all
447  // functions
448  goto_programt tmp;
449  tmp.add_instruction();
450  goto_programt::const_targett sh_target=tmp.instructions.begin();
451  generate_state(sh_target);
452  statet &shared_state=get_state(sh_target);
453 
454  struct wl_entryt
455  {
456  wl_entryt(
457  const irep_idt &_function_identifier,
458  const goto_programt &_goto_program,
459  locationt _location)
460  : function_identifier(_function_identifier),
461  goto_program(&_goto_program),
462  location(_location)
463  {
464  }
465 
466  irep_idt function_identifier;
467  const goto_programt *goto_program;
468  locationt location;
469  };
470

472  thread_wlt thread_wl;
473 
474  for(const auto &gf_entry : goto_functions.function_map)
475  {
476  forall_goto_program_instructions(t_it, gf_entry.second.body)
477  {
478  if(is_threaded(t_it))
479  {
480  thread_wl.push_back(
481  wl_entryt(gf_entry.first, gf_entry.second.body, t_it));
482

488  merge_shared(shared_state, end_state, sh_target);
489  }
490  }
491  }
492 
493  // new feed in the shared state into all concurrently executing
494  // functions, and iterate until the shared state stabilizes
495  bool new_shared=true;
496  while(new_shared)
497  {
498  new_shared=false;
499 
500  for(const auto &thread : thread_wl)
501  {
502  working_sett working_set;
503  put_in_working_set(working_set, thread.location);
504 
505  statet &begin_state = get_state(thread.location);
506  merge(begin_state, shared_state, thread.location);
507 
508  while(!working_set.empty())
509  {
510  goto_programt::const_targett l=get_next(working_set);
511 
512  visit(
513  thread.function_identifier,
514  l,
515  working_set,
516  *thread.goto_program,
517  goto_functions);
518 
519  // the underlying domain must make sure that the final state
520  // carries all possible values; otherwise we would need to
521  // merge over each and every state
522  if(l->is_end_function())
523  {
524  statet &end_state=get_state(l);

526  }
